$\text{\color{gray} {// Построение двоичного кода натурального числа}} $

$\text{{\textbf{Вход:} натуральное число }} {{k : 0 \leq k \leq 2^n - 1.}} $

$\text{{\textbf{Выход:} битовая шкала }} {{B_{n.}}} $

$\text{\color{blue} \textbf{func }}B(k\ :\ \text{\color{blue} \textbf{int}},\ n\ :\ \text{\color{blue} \textbf{int}}) :\ \text{\color{blue} \textbf{array }}[1..n]\text{\color{blue} \textbf{ of }}0..1$

$\quad B_{n}\ :\ \text{\color{blue} \textbf{array }}[1..n]\text{\color{blue} \textbf{ of }}0..1\ $

$\quad \ $

$\quad \text{\color{blue} \textbf{for }}i\text{\color{blue} \textbf{ from }}n\text{\color{blue} \textbf{ downto }}1\text{\color{blue} \textbf{ do }}$

$\quad \quad B_{n}[i] := k\text{\color{blue} \textbf{ mod }}2\ \text{\color{gray} {// очередной бит}} $

$\quad \quad k := k\text{\color{blue} \textbf{ div }}2\ \text{\color{gray} {// оставшееся число}} $

$\quad \text{\color{blue} \textbf{end for}}$

$\quad \ $

$\quad \text{\color{blue} \textbf{return}}\ B_{n}\ $

$\text{\color{blue} \textbf{end func }}$

$\text{\color{blue} \textbf{func }}B_{all}(n\ :\ \text{\color{blue} \textbf{int}}) $

$\quad \text{\color{blue} \textbf{for }}i\text{\color{blue} \textbf{ from }}0\text{\color{blue} \textbf{ to }}2 ^ n\ -\ 1\text{\color{blue} \textbf{ do }}\text{\color{blue} \textbf{yield }}B(i,\ n)\ \text{\color{blue} \textbf{end for}}$

$\text{\color{blue} \textbf{end func }}$

$\text{{\textbf{Вход:} }} {{i -}} \text{{ номер подмножества}} $

$\text{{\textbf{Выход:} номер изменяемого разряда}} $

$\text{\color{blue} \textbf{func }}Q(i\ :\ \text{\color{blue} \textbf{int}}) :\ \text{\color{blue} \textbf{int}}$

$\quad q := 1;\ j := i\ $

$\quad \text{\color{blue} \textbf{while }}j\text{\color{blue} \textbf{ mod }}2\ =\ 0\text{\color{blue} \textbf{ do }}j := j\text{\color{blue} \textbf{ div }}2;\ q := q\ +\ 1\ \text{\color{blue} \textbf{end while}}$

$\quad \text{\color{blue} \textbf{return}}\ q\ $

$\text{\color{blue} \textbf{end func }}$

$\text{\color{gray} {// Построение бинарного кода Грея }} {\color{gray} {n-}} \text{\color{gray} {элементного множества}} $

$\text{{\textbf{Вход:} число }} {{n \geq 0 -}} \text{{ мощность множества}} $

$\text{{\textbf{Выход:} последовательность кодов подмножеств }} {{B}} $

$\text{\color{blue} \textbf{func }}Gray(n\ :\ \text{\color{blue} \textbf{int}}) $

$\quad B_{n}\ :\ \text{\color{blue} \textbf{array }}[1..n]\text{\color{blue} \textbf{ of }}0..1\ \text{\color{gray} {// битовая шкала}} $

$\quad \text{\color{blue} \textbf{for }}i\text{\color{blue} \textbf{ from }}1\text{\color{blue} \textbf{ to }}n\text{\color{blue} \textbf{ do }}B_{n}[i] := 0\ \text{\color{blue} \textbf{end for}} \text{\color{gray} {// инициализация}} $

$\quad \text{\color{blue} \textbf{yield }}B_{n}\ \text{\color{gray} {// пустое множество}} $

$\quad \text{\color{blue} \textbf{for }}i\text{\color{blue} \textbf{ from }}1\text{\color{blue} \textbf{ to }}2 ^ n\ -\ 1\text{\color{blue} \textbf{ do }}$

$\quad \quad p := Q(i)\ \text{\color{gray} {// определение номера элемента}} $

$\quad \quad B_{n}[p] := 1\ -\ B_{n}[p]\ \text{\color{gray} {// добавление или удаление элемента}} $

$\quad \quad \text{\color{blue} \textbf{yield }}B_{n}\ \text{\color{gray} {// очередное подмножество}} $

$\quad \text{\color{blue} \textbf{end for}}$

$\text{\color{blue} \textbf{end func }}$
